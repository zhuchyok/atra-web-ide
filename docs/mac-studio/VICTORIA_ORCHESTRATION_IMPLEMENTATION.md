async def _collect_expert_responses(self, experts: List[Tuple[str, Dict]], goal: str) -> List[Dict]:
    """–°–æ–±—Ä–∞—Ç—å –æ—Ç–≤–µ—Ç—ã –æ—Ç —ç–∫—Å–ø–µ—Ä—Ç–æ–≤ –ø–∞—Ä–∞–ª–ª–µ–ª—å–Ω–æ"""
    tasks = []
    for expert_name, expert_data in experts:
        task = self._execute_via_expert(expert_name, expert_data, goal)
        tasks.append(task)
    
    responses = await asyncio.gather(*tasks, return_exceptions=True)
    
    # –§–æ—Ä–º–∏—Ä—É–µ–º —Å–ø–∏—Å–æ–∫ –æ—Ç–≤–µ—Ç–æ–≤
    result = []
    for i, response in enumerate(responses):
        if isinstance(response, Exception):
            logger.error(f"–û—à–∏–±–∫–∞ –æ—Ç —ç–∫—Å–ø–µ—Ä—Ç–∞ {experts[i][0]}: {response}")
            result.append({
                "expert": experts[i][0],
                "response": f"–û—à–∏–±–∫–∞: {response}",
                "error": True
            })
        else:
            result.append({
                "expert": experts[i][0],
                "response": str(response),
                "error": False
            })
    
    return result

    pool = await self._get_db_pool()
    if not pool:
        return None
    
    try:
        async with pool.acquire() as conn:
            experts = await conn.fetch("""
                SELECT name, role, system_prompt, department
                FROM experts
                WHERE department = $1
            """, department)
            
            if not experts:
                return None
            
            # –û—Ü–µ–Ω–∏–≤–∞–µ–º –∫–∞–∂–¥–æ–≥–æ —ç–∫—Å–ø–µ—Ä—Ç–∞ –æ—Ç–¥–µ–ª–∞
            best_expert = None
            best_score = -1
            
            for expert in experts:
                score = 0.0
                
                # –†–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ—Å—Ç—å —Ä–æ–ª–∏
                role = expert['role'].lower()
                goal_lower = goal.lower()
                if any(kw in role for kw in goal_lower.split()[:3]):

                if expert_id:
                    completed = await conn.fetchval("""
                        SELECT COUNT(*) FROM tasks
                        WHERE assignee_expert_id = $1 AND status = 'completed'
                    """, expert_id) or 0
                    score += completed * 0.5
                
                if score > best_score:
                    best_score = score
                    best_expert = (expert['name'], {
                        'role': expert['role'],
                        'system_prompt': expert['system_prompt'],
                        'department': expert['department']
                    })
            
            return best_expert
