```python
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –¥–æ—Å—Ç—É–ø–Ω—ã–µ —Å—Ä–µ–¥—Å—Ç–≤–∞ –¥–ª—è —Ç–æ—Ä–≥–æ–≤–ª–∏."""

    # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –∑–∞–Ω—è—Ç—ã–µ —Å—Ä–µ–¥—Å—Ç–≤–∞
    total_used = 0.0
    for position in user_data.get("positions", []):
        if position.get("status") == "open":
            if trade_mode == "futures":
                # –î–ª—è —Ñ—å—é—á–µ—Ä—Å–æ–≤ - –º–∞—Ä–∂–∞
                total_used += float(position.get("margin", 0))

    return max(0, deposit - total_used)
```
