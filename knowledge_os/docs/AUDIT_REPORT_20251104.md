def calculate_crypto_metrics(trades: List[Dict], risk_free_rate: float = 0.02):
    """
    –†–∞—Å—á–µ—Ç –º–µ—Ç—Ä–∏–∫ –¥–ª—è –∫—Ä–∏–ø—Ç–æ—Ä—ã–Ω–∫–∞ (24/7, –Ω–µ—Ç –≤—ã—Ö–æ–¥–Ω—ã—Ö)

    # Sortino Ratio (—Ç–æ–ª—å–∫–æ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–µ –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç–∏)
    negative_returns = daily_returns[daily_returns < 0]
    downside_vol = negative_returns.std() * np.sqrt(365) if len(negative_returns) > 0 else 0
    sortino_ratio = (annual_return - risk_free_rate) / downside_vol if downside_vol > 0 else 0
    
    # Max Drawdown
    cumulative = (1 + daily_returns).cumprod()
    rolling_max = cumulative.expanding().max()
    drawdown = (cumulative - rolling_max) / rolling_max
    max_drawdown = abs(drawdown.min()) * 100
    
    return {
        'sharpe_ratio': sharpe_ratio,
        'sortino_ratio': sortino_ratio,
        'max_drawdown_pct': max_drawdown,
        'annual_return_pct': annual_return * 100,
        'volatility_pct': volatility * 100
    }
```

def calculate_adaptive_tp_sl(atr: float, base_tp: float, base_sl: float, volatility_regime: str) -> Tuple[float, float]:
    """
    –ê–¥–∞–ø—Ç–∞—Ü–∏—è TP/SL –∫ –≤–æ–ª–∞—Ç–∏–ª—å–Ω–æ—Å—Ç–∏
    
    Args:
        atr: Average True Range
        base_tp: –ë–∞–∑–æ–≤—ã–π TP (–ø—Ä–æ—Ü–µ–Ω—Ç)
        base_sl: –ë–∞–∑–æ–≤—ã–π SL (–ø—Ä–æ—Ü–µ–Ω—Ç)
        volatility_regime: 'LOW', 'NORMAL', 'HIGH', 'EXTREME'
    
    Returns:
        (tp, sl) –∞–¥–∞–ø—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —É—Ä–æ–≤–Ω–∏
    """
    multipliers = {
        'LOW': {'tp': 0.8, 'sl': 0.9},      # –£–∑–∫–∏–µ —É—Ä–æ–≤–Ω–∏ –≤ –Ω–∏–∑–∫–æ–π –≤–æ–ª–∞—Ç–∏–ª—å–Ω–æ—Å—Ç–∏
        'NORMAL': {'tp': 1.0, 'sl': 1.0},    # –°—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã–µ —É—Ä–æ–≤–Ω–∏
        'HIGH': {'tp': 1.3, 'sl': 1.2},      # –®–∏—Ä–æ–∫–∏–µ —É—Ä–æ–≤–Ω–∏ –≤ –≤—ã—Å–æ–∫–æ–π –≤–æ–ª–∞—Ç–∏–ª—å–Ω–æ—Å—Ç–∏
        'EXTREME': {'tp': 1.5, 'sl': 1.5}    # –û—á–µ–Ω—å —à–∏—Ä–æ–∫–∏–µ —É—Ä–æ–≤–Ω–∏
    }
    
    mult = multipliers.get(volatility_regime, multipliers['NORMAL'])
    
    # –ê–¥–∞–ø—Ç–∏—Ä—É–µ–º –∫ ATR
    atr_multiplier = atr / 100  # –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º ATR
    tp = base_tp * mult['tp'] * (1 + atr_multiplier)
    sl = base_sl * mult['sl'] * (1 + atr_multiplier)
    
    return tp, sl
```
