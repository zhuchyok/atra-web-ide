def get_adaptive_position_limit(user_data):
    """
    –ü–æ–ª—É—á–∞–µ—Ç –∞–¥–∞–ø—Ç–∏–≤–Ω—ã–π –ª–∏–º–∏—Ç –ø–æ–∑–∏—Ü–∏–π –¥–ª—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è

    if not ADAPTIVE_POSITION_CONFIG["enabled"]:
        return 8  # –§–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –ª–∏–º–∏—Ç

    balance = user_data.get("deposit", 0)

    position_limit = ADAPTIVE_POSITION_CONFIG["min_positions"]
    for threshold, limit in thresholds.items():
        if threshold == "max":
            if balance >= 20000:
                position_limit = limit
        elif balance < threshold:
            position_limit = limit
            break

    # –ü—Ä–æ–≤–µ—Ä—è–µ–º –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–µ –ø–æ –æ–±—â–µ–º—É —Ä–∏—Å–∫—É
    risk_per_trade = user_data.get("risk_pct", 2.0) / 100
    max_total_risk = balance * ADAPTIVE_POSITION_CONFIG["max_total_risk_pct"] / 100
    max_positions_by_risk = int(max_total_risk / (balance * risk_per_trade))

    return min(position_limit, max_positions_by_risk)
